import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Load employee data (replace with CSV if available)
np.random.seed(7)
departments = ["Marketing", "R&D", "IT", "Operations", "Finance", "HR", "Sales", "Customer Support"]
regions = ["Africa", "Latin America", "North America", "Asia Pacific", "Europe", "Middle East"]
n = 100
df = pd.DataFrame({
    "employee_id": [f"EMP{str(i).zfill(3)}" for i in range(1, n+1)],
    "department": np.random.choice(departments, size=n, p=[0.10,0.13,0.16,0.14,0.12,0.10,0.15,0.10]),
    "region": np.random.choice(regions, size=n),
    "performance_score": np.round(np.random.normal(loc=75, scale=12, size=n).clip(40, 100), 2),
    "years_experience": np.random.randint(0, 21, size=n),
    "satisfaction_rating": np.round(np.random.uniform(2.5, 5.0, size=n), 1),
})

# Frequency of Finance department
finance_freq = (df["department"] == "Finance").sum()
print("Finance department frequency count:", finance_freq)

# Create histogram of department distribution
counts = df["department"].value_counts().sort_index()
plt.figure(figsize=(8,5))
plt.bar(counts.index, counts.values)
plt.title("Department Distribution (n=100)")
plt.xlabel("Department")
plt.ylabel("Number of Employees")
plt.xticks(rotation=30, ha="right")
plt.tight_layout()
plt.savefig("department_distribution.png")  # Save the histogram
plt.show()
